export default function UploadPage() {

  const queryClient = useQueryClient();
  const { isAuthenticated, user } = useAuth();

        body: data,
        credentials: 'include',

        throw new Error(`${response.status}: ${errorText}`);

      return response.json();

      });

// Funnel Creation Modal Component
function FunnelCreationModal({ 
  contentId, 
  funnelName, 
  onComplete, 
  onSkip 
}: {
  contentId: number | null;
  funnelName: string;
  onComplete: () => void;
  onSkip: () => void;
}) {
  const { toast } = useToast();
  const [currentStep, setCurrentStep] = useState(0);
  const [funnelId, setFunnelId] = useState<string | null>(null);
  const [upsellSteps, setUpsellSteps] = useState([
    { 
      title: "", 
      description: "", 
      price: "", 
      pricingModel: "fixed" as "fixed" | "tip",
      minimumPrice: "",
      category: "",
      accessType: "download" as "download" | "view" | "stream" | "link",
      ctaText: "Add this to my order",
      offerPriceId: "", // Will be auto-generated
      // Content fields
      file: null as File | null,
      fileName: "",
      previewImage: "",
      linkUrl: "",
      textContent: ""
    }
  ]);

  // Create funnel mutation
  const createFunnelMutation = useMutation({
    mutationFn: async () => {
      const response = await fetch('/api/funnels', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({ name: funnelName }),
      });
      if (!response.ok) throw new Error('Failed to create funnel');
      return response.json();
    },
    onSuccess: (data) => {
      setFunnelId(data.id);
      attachProductToFunnel(data.id);
    },
  });

  // Attach product to funnel
  const attachProductToFunnel = async (funnelId: string) => {
    try {
      const response = await fetch(`/api/funnels/${funnelId}/products`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({ contentIds: [contentId] }),
      });
      if (!response.ok) throw new Error('Failed to attach product');
      setCurrentStep(1);
    } catch (error) {
      toast({
        title: "Error",
        description: "Failed to attach product to funnel",
        variant: "destructive",
      });
    }
  };

  // Add upsell step mutation
  const addStepMutation = useMutation({
    mutationFn: async (stepData: any) => {
      const response = await fetch(`/api/funnels/${funnelId}/steps`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({
          ...stepData,
          stepOrder: upsellSteps.findIndex(s => s === stepData) + 1,
        }),
      });
      if (!response.ok) throw new Error('Failed to add step');
      return response.json();
    },
  });

  const handleCreateFunnel = () => {
    createFunnelMutation.mutate();
  };

  const handleAddStep = async (stepIndex: number) => {
    const step = upsellSteps[stepIndex];
    if (!step.title || !step.price) {
      toast({
        title: "Error",
        description: "Please fill in title and price",
        variant: "destructive",
      });
      return;
    }

    // Validate content based on access type
    if (step.accessType === 'download' && !step.file) {
      toast({
        title: "Error",
        description: "Please upload a file for download products",
        variant: "destructive",
      });
      return;
    }

    if (step.accessType === 'link' && !step.linkUrl) {
      toast({
        title: "Error", 
        description: "Please provide a link URL",
        variant: "destructive",
      });
      return;
    }

    try {
      // First upload the content if it's a file-based product
      let contentData = null;
      if (step.file) {
        const formData = new FormData();
        formData.append('file', step.file);
        formData.append('title', step.title);
        formData.append('description', step.description);
        formData.append('price', step.price);
        formData.append('pricingModel', step.pricingModel);
        if (step.minimumPrice) formData.append('minimumPrice', step.minimumPrice);
        if (step.category) formData.append('category', step.category);
        formData.append('accessType', step.accessType);
        if (step.previewImage) formData.append('previewImage', step.previewImage);
        if (step.linkUrl) formData.append('linkUrl', step.linkUrl);
        if (step.textContent) formData.append('textContent', step.textContent);

        const uploadResponse = await fetch('/api/content', {
          method: 'POST',
          credentials: 'include',
          body: formData,
        });

        if (!uploadResponse.ok) {
          throw new Error('Failed to upload upsell content');
        }

        contentData = await uploadResponse.json();
      }

      // Create Stripe product and price
      const stripeResponse = await fetch('/api/stripe/create-upsell-product', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({
          name: step.title,
          description: step.description,
          price: parseFloat(step.price),
        }),
      });

      if (!stripeResponse.ok) {
        throw new Error('Failed to create Stripe product');
      }

      const { priceId } = await stripeResponse.json();

      // Add the upsell step with content reference
      await addStepMutation.mutateAsync({
        title: step.title,
        description: step.description,
        offerPriceId: priceId,
        ctaText: step.ctaText,
        contentId: contentData?.id || null,
      });

      toast({
        title: "Success",
        description: "Upsell step added successfully!",
      });
    } catch (error) {
      toast({
        title: "Error",
        description: "Failed to add upsell step",
        variant: "destructive",
      });
    }
  };

  const addUpsellStep = () => {
    setUpsellSteps([
      ...upsellSteps,
      { 
        title: "", 
        description: "", 
        price: "", 
        pricingModel: "fixed" as "fixed" | "tip",
        minimumPrice: "",
        category: "",
        accessType: "download" as "download" | "view" | "stream" | "link",
        ctaText: "Add this to my order",
        offerPriceId: "",
        file: null,
        fileName: "",
        previewImage: "",
        linkUrl: "",
        textContent: ""
      }
    ]);
  };

  const removeUpsellStep = (index: number) => {
    setUpsellSteps(upsellSteps.filter((_, i) => i !== index));
  };

  const updateUpsellStep = (index: number, field: string, value: any) => {
    const updated = [...upsellSteps];
    updated[index] = { ...updated[index], [field]: value };
    setUpsellSteps(updated);
  };

  return (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-4">
      <Card className="w-full max-w-2xl glassmorphism border-white/10 max-h-[90vh] overflow-y-auto">
        <CardContent className="p-6">
          <div className="flex items-center justify-between mb-6">
            <h2 className="text-2xl font-bold bg-gradient-to-r from-purple-400 to-pink-400 bg-clip-text text-transparent">
              Create Upsell Funnel
            </h2>
            <Button
              variant="ghost"
              size="sm"
              onClick={onSkip}
              className="text-gray-400 hover:text-white"
            >
              <X className="w-5 h-5" />
            </Button>
          </div>

          {currentStep === 0 && (
            <div className="space-y-6">
              <div className="text-center">
                <Target className="w-16 h-16 text-purple-400 mx-auto mb-4" />
                <h3 className="text-xl font-semibold mb-2">Setup Your Upsell Funnel</h3>
                <p className="text-gray-300">
                  Create "{funnelName}" and start maximizing your revenue with strategic upsells.
                </p>
              </div>

              <div className="flex space-x-4">
                <Button
                  variant="outline"
                  onClick={onSkip}
                  className="flex-1 border-white/20"
                >
                  Skip for Now
                </Button>
                <Button
                  onClick={handleCreateFunnel}
                  disabled={createFunnelMutation.isPending}
                  className="flex-1 bg-purple-500 hover:bg-purple-600"
                >
                  {createFunnelMutation.isPending ? "Creating..." : "Create Funnel"}
                </Button>
              </div>
            </div>
          )}

          {currentStep === 1 && (
            <div className="space-y-6">
              <div>
                <h3 className="text-xl font-semibold mb-2">Add Upsell Offers</h3>
                <p className="text-gray-300">
                  Create offers that will be shown to customers after they purchase your main product.
                </p>
              </div>

              <div className="space-y-4">
                {upsellSteps.map((step, index) => (
                  <Card key={index} className="bg-white/5 border-white/10">
                    <CardContent className="p-4">
                      <div className="flex items-center justify-between mb-4">
                        <h4 className="font-semibold">Upsell Step {index + 1}</h4>
                        {upsellSteps.length > 1 && (
                          <Button
                            variant="ghost"
                            size="sm"
                            onClick={() => removeUpsellStep(index)}
                            className="text-red-400 hover:text-red-300"
                          >
                            <X className="w-4 h-4" />
                          </Button>
                        )}
                      </div>

                      {/* Basic Info */}
                      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                        <div>
                          <label className="block text-sm font-medium mb-1">Title *</label>
                          <Input
                            value={step.title}
                            onChange={(e) => updateUpsellStep(index, 'title', e.target.value)}
                            placeholder="e.g., Bonus Training Module"
                            className="bg-white/10 border-white/20"
                          />
                        </div>
                        <div>
                          <label className="block text-sm font-medium mb-1">Category</label>
                          <Input
                            value={step.category}
                            onChange={(e) => updateUpsellStep(index, 'category', e.target.value)}
                            placeholder="e.g., Digital Products"
                            className="bg-white/10 border-white/20"
                          />
                        </div>
                      </div>

                      <div className="mt-4">
                        <label className="block text-sm font-medium mb-1">Description</label>
                        <Textarea
                          value={step.description}
                          onChange={(e) => updateUpsellStep(index, 'description', e.target.value)}
                          placeholder="Describe this upsell offer..."
                          className="bg-white/10 border-white/20"
                          rows={3}
                        />
                      </div>

                      {/* Access Type */}
                      <div className="mt-4">
                        <label className="block text-sm font-medium mb-2">Access Type *</label>
                        <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
                          {[
                            { key: 'download', label: 'Download', icon: Download, desc: 'File download' },
                            { key: 'view', label: 'View', icon: Eye, desc: 'Text/content view' },
                            { key: 'stream', label: 'Stream', icon: Play, desc: 'Media streaming' },
                            { key: 'link', label: 'Link', icon: ExternalLink, desc: 'External redirect' },
                          ].map(({ key, label, icon: Icon, desc }) => (
                            <button
                              key={key}
                              type="button"
                              onClick={() => updateUpsellStep(index, 'accessType', key)}
                              className={`p-3 rounded-lg border transition-all text-left ${
                                step.accessType === key
                                  ? 'border-purple-400 bg-purple-500/20'
                                  : 'border-white/20 bg-white/5 hover:bg-white/10'
                              }`}
                            >
                              <Icon className="w-5 h-5 mb-2" />
                              <div className="font-medium text-sm">{label}</div>
                              <div className="text-xs text-gray-400">{desc}</div>
                            </button>
                          ))}
                        </div>
                      </div>

                      {/* File Upload for Download/View/Stream */}
                      {(step.accessType === 'download' || step.accessType === 'view' || step.accessType === 'stream') && (
                        <div className="mt-4">
                          <label className="block text-sm font-medium mb-2">Upload File *</label>
                          <div 
                            className="border-2 border-dashed border-white/20 rounded-lg p-6 text-center hover:border-purple-400 transition-colors"
                            onDragOver={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.add('border-purple-400', 'bg-purple-500/10');
                            }}
                            onDragLeave={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.remove('border-purple-400', 'bg-purple-500/10');
                            }}
                            onDrop={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.remove('border-purple-400', 'bg-purple-500/10');
                              const file = e.dataTransfer.files[0];
                              if (file) {
                                updateUpsellStep(index, 'file', file);
                                updateUpsellStep(index, 'fileName', file.name);
                              }
                            }}
                          >
                            <input
                              type="file"
                              accept="*/*"
                              onChange={(e) => {
                                const file = e.target.files?.[0];
                                if (file) {
                                  updateUpsellStep(index, 'file', file);
                                  updateUpsellStep(index, 'fileName', file.name);
                                }
                              }}
                              className="hidden"
                              id={`file-${index}`}
                            />
                            <label htmlFor={`file-${index}`} className="cursor-pointer block">
                              {step.file ? (
                                <div className="text-center">
                                  <FileUp className="w-8 h-8 text-green-400 mx-auto mb-2" />
                                  <p className="text-green-400 font-medium">{step.fileName}</p>
                                  <p className="text-xs text-gray-400 mt-1">Click or drag to change file</p>
                                </div>
                              ) : (
                                <div className="text-center">
                                  <Upload className="w-8 h-8 text-gray-400 mx-auto mb-2" />
                                  <p className="text-gray-400">Click or drag to upload file</p>
                                  <p className="text-xs text-gray-500 mt-1">Any file type supported</p>
                                </div>
                              )}
                            </label>
                          </div>
                        </div>
                      )}

                      {/* Link URL for Link type */}
                      {step.accessType === 'link' && (
                        <div className="mt-4">
                          <label className="block text-sm font-medium mb-1">Link URL *</label>
                          <Input
                            value={step.linkUrl}
                            onChange={(e) => updateUpsellStep(index, 'linkUrl', e.target.value)}
                            placeholder="https://example.com"
                            className="bg-white/10 border-white/20"
                          />
                        </div>
                      )}

                      {/* Text Content for View type */}
                      {step.accessType === 'view' && (
                        <div className="mt-4">
                          <label className="block text-sm font-medium mb-1">Text Content</label>
                          <Textarea
                            value={step.textContent}
                            onChange={(e) => updateUpsellStep(index, 'textContent', e.target.value)}
                            placeholder="Enter the text content customers will see..."
                            className="bg-white/10 border-white/20"
                            rows={5}
                          />
                        </div>
                      )}

                      {/* Preview Image */}
                      <div className="mt-4">
                        <label className="block text-sm font-medium mb-2">Preview Image</label>
                        <div className="flex items-center space-x-4">
                          <div
                            className="border-2 border-dashed border-white/20 rounded-lg p-4 hover:border-purple-400 transition-colors"
                            onDragOver={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.add('border-purple-400', 'bg-purple-500/10');
                            }}
                            onDragLeave={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.remove('border-purple-400', 'bg-purple-500/10');
                            }}
                            onDrop={(e) => {
                              e.preventDefault();
                              e.currentTarget.classList.remove('border-purple-400', 'bg-purple-500/10');
                              const file = e.dataTransfer.files[0];
                              if (file && file.type.startsWith('image/')) {
                                const reader = new FileReader();
                                reader.onload = (event) => {
                                  updateUpsellStep(index, 'previewImage', event.target?.result as string);
                                };
                                reader.readAsDataURL(file);
                              }
                            }}
                          >
                            <input
                              type="file"
                              accept="image/*"
                              onChange={(e) => {
                                const file = e.target.files?.[0];
                                if (file) {
                                  const reader = new FileReader();
                                  reader.onload = (event) => {
                                    updateUpsellStep(index, 'previewImage', event.target?.result as string);
                                  };
                                  reader.readAsDataURL(file);
                                }
                              }}
                              className="hidden"
                              id={`preview-${index}`}
                            />
                            <label
                              htmlFor={`preview-${index}`}
                              className="px-4 py-2 bg-white/10 border border-white/20 rounded-lg cursor-pointer hover:bg-white/20 transition-colors block text-center"
                            >
                              <Camera className="w-4 h-4 inline mr-2" />
                              Click or drag image
                            </label>
                          </div>
                          {step.previewImage && (
                            <div className="relative">
                              <img
                                src={step.previewImage}
                                alt="Preview"
                                className="w-16 h-16 object-cover rounded-lg"
                              />
                              <button
                                type="button"
                                onClick={() => updateUpsellStep(index, 'previewImage', '')}
                                className="absolute -top-2 -right-2 w-6 h-6 bg-red-500 text-white rounded-full flex items-center justify-center"
                              >
                                <X className="w-3 h-3" />
                              </button>
                            </div>
                          )}
                        </div>
                      </div>

                      {/* Pricing */}
                      <div className="mt-4">
                        <label className="block text-sm font-medium mb-2">Pricing Model</label>
                        <div className="grid grid-cols-2 gap-3">
                          <button
                            type="button"
                            onClick={() => updateUpsellStep(index, 'pricingModel', 'fixed')}
                            className={`p-3 rounded-lg border transition-all ${
                              step.pricingModel === 'fixed'
                                ? 'border-purple-400 bg-purple-500/20'
                                : 'border-white/20 bg-white/5 hover:bg-white/10'
                            }`}
                          >
                            <DollarSign className="w-5 h-5 mb-1" />
                            <div className="font-medium text-sm">Fixed Price</div>
                            <div className="text-xs text-gray-400">Set exact price</div>
                          </button>
                          <button
                            type="button"
                            onClick={() => updateUpsellStep(index, 'pricingModel', 'tip')}
                            className={`p-3 rounded-lg border transition-all ${
                              step.pricingModel === 'tip'
                                ? 'border-purple-400 bg-purple-500/20'
                                : 'border-white/20 bg-white/5 hover:bg-white/10'
                            }`}
                          >
                            <Heart className="w-5 h-5 mb-1" />
                            <div className="font-medium text-sm">Pay-What-You-Want</div>
                            <div className="text-xs text-gray-400">Let customers tip</div>
                          </button>
                        </div>
                      </div>

                      <div className="mt-4 grid grid-cols-1 md:grid-cols-2 gap-4">
                        <div>
                          <label className="block text-sm font-medium mb-1">
                            {step.pricingModel === 'tip' ? 'Minimum Price *' : 'Price *'}
                          </label>
                          <div className="relative">
                            <span className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400">$</span>
                            <Input
                              type="number"
                              step="0.01"
                              min="0.50"
                              value={step.price}
                              onChange={(e) => updateUpsellStep(index, 'price', e.target.value)}
                              placeholder="29.99"
                              className="bg-white/10 border-white/20 pl-8"
                            />
                          </div>
                        </div>
                        <div>
                          <label className="block text-sm font-medium mb-1">Call-to-Action Text</label>
                          <Input
                            value={step.ctaText}
                            onChange={(e) => updateUpsellStep(index, 'ctaText', e.target.value)}
                            placeholder="Add this to my order"
                            className="bg-white/10 border-white/20"
                          />
                        </div>
                      </div>

                      <div className="mt-4 flex justify-end">
                        <Button
                          onClick={() => handleAddStep(index)}
                          disabled={addStepMutation.isPending}
                          size="sm"
                          className="bg-purple-500/50 hover:bg-purple-500/70"
                        >
                          {addStepMutation.isPending ? "Adding..." : "Add Step"}
                        </Button>
                      </div>
                    </CardContent>
                  </Card>
                ))}

                <Button
                  onClick={addUpsellStep}
                  variant="outline"
                  className="w-full border-white/20 border-dashed"
                >
                  <Plus className="w-4 h-4 mr-2" />
                  Add Another Upsell
                </Button>
              </div>

              <div className="flex space-x-4">
                <Button
                  variant="outline"
                  onClick={onComplete}
                  className="flex-1 border-white/20"
                >
                  Finish Later
                </Button>
                <Button
                  onClick={onComplete}
                  className="flex-1 bg-green-500 hover:bg-green-600"
                >
                  Complete Setup
                </Button>
              </div>
            </div>
          )}
        </CardContent>
      </Card>
    </div>
  );
}
