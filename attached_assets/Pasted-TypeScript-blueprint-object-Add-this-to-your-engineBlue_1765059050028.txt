TypeScript blueprint object
Add this to your engineBlueprints.ts:
export const youtubeMetricsBlueprint = {
  // Identity
  id: "youtube-metrics-engine",
  engineId: "calculator", // or "ai-analysis" later if you add analysis logic
  name: "YouTube Metrics Engine",
  description:
    "Generates YouTube metric calculators for CTR, CPM, retention, and more across longform and Shorts.",
  segment: "creator", // creator | social | mbb

  // Slug & Content Patterns
  // youtube-ctr-longform-calculator
  // youtube-retention-shorts-calculator
  slugPattern: {
    template: "youtube-{metric}-{content_type}-calculator",
  },

  // YouTube CTR Longform Calculator
  // YouTube Retention Shorts Calculator
  titlePattern: "YouTube {metric_label} {content_type_label} Calculator",

  // Primary SEO keyword
  // "youtube ctr calculator", "youtube shorts retention calculator", etc.
  keywordPattern: "youtube {metric} {content_type} calculator",

  // Meta / description text
  descriptionPattern:
    "Calculate your YouTube {metric_label} for {content_type_label} videos and understand how your channel is performing.",

  // DIMENSIONS – the axes of the Cartesian product
  dimensions: [
    {
      id: "metric",
      label: "Metric",
      values: [
        { id: "ctr", label: "CTR" },
        { id: "cpm", label: "CPM" },
        { id: "retention", label: "Retention" },
        { id: "avg-view-duration", label: "Average View Duration" },
        { id: "clicks-per-impression", label: "Clicks per Impression" },
      ],
    },
    {
      id: "content_type",
      label: "Content Type",
      values: [
        { id: "longform", label: "Longform" },
        { id: "shorts", label: "Shorts" },
      ],
    },
  ],
  // → 5 metrics × 2 content_types = 10 shells

  // Cluster assignment (for internal linking)
  clusterResolver: (combo: { metric: string; content_type: string }) => {
    // All YouTube metrics go under a single YouTube Metrics cluster
    return "youtube-metrics";
  },

  // Link rules for SEO/internal linking
  linkRules: {
    // Each tool should link to ~3 sibling metrics in the same cluster
    siblingsPerTool: 3,

    // Each tool should be associated with ~2 articles in the YouTube metrics cluster
    // e.g. "What Is a Good YouTube CTR?", "How to Improve YouTube Retention"
    articlesPerTool: 2,

    // Pillar page that acts as the hub for all these calculators
    pillarSlug: "youtube-metrics",

    defaultCTAs: {
      quickCTA:
        "Use this calculator to understand your YouTube {metric_label} for {content_type_label} videos, then explore other YouTube metrics tools below.",
      bottomCTA:
        "Ready to go deeper? Try the other YouTube metrics calculators on this page to complete your analytics picture.",
    },
  },

  // Defaults for generated shells
  defaults: {
    priority: "experimental", // or "normal" if you want them higher priority
    isIndexed: false, // you’ll turn this on per-tool after audit
    inDirectory: false, // keep them out of /tools until you decide
    searchIntent: "action", // user wants to calculate something
  },

  // IO descriptor – adjust to match your internal enums
  inputType: "form", // user enters impressions, clicks, watch time, etc.
  outputType: "stats", // calculator returns computed metrics
};
